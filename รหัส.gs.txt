function initSheet() {
  throw new Error('Safety stop: initSheet() rebuilds the spreadsheet and can delete existing stock data. Use initSheet_CONFIRMED_RESET() only for a brand-new empty Sheet.');
}

function initSheet_CONFIRMED_RESET() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheets = ss.getSheets();
  const keeper = sheets[0];

  for (let i = 1; i < sheets.length; i++) {
    ss.deleteSheet(sheets[i]);
  }

  const wsDash = buildDashboard(ss, keeper);
  const wsDev  = buildDeviceSheet(ss);
  const wsMask = buildMaskSheet(ss);
  const wsAcc  = buildAccessorySheet(ss);
  const wsHist = buildHistorySheet(ss);
  const wsProd = buildProductDBSheet(ss);

  wsDash.setTabColor('#1a1a2e');
  wsDev.setTabColor('#1a4fa0');
  wsMask.setTabColor('#1a7a4a');
  wsAcc.setTabColor('#a06a00');
  wsHist.setTabColor('#4a3580');
  wsProd.setTabColor('#555555');

  ss.setActiveSheet(wsDash);
  SpreadsheetApp.flush();
}

function buildDashboard(ss, existingSheet) {
  const ws = existingSheet;
  ws.setName(TABS.dashboard);
  ws.setHiddenGridlines(true);
  ws.clearContents().clearFormats();

  ws.setRowHeight(1, 50);
  ws.getRange('A1:I1').merge()
    .setValue('SKYLINE MEDICAL — Stock Management')
    .setBackground('#1a1a2e')
    .setFontFamily('Arial').setFontSize(16).setFontWeight('bold').setFontColor('#FFFFFF')
    .setHorizontalAlignment('center').setVerticalAlignment('middle');

  ws.getRange('A2:I2').merge()
    .setValue('Auto-updated via Apps Script')
    .setFontFamily('Arial').setFontSize(10).setFontColor('#888888')
    .setHorizontalAlignment('center');
  ws.setRowHeight(2, 18);

  [4, 5, 6, 7, 8].forEach(r => ws.setRowHeight(r, 22));
  ws.setRowHeight(6, 36);

  const cards = [
    {
      col:'B',
      label:'CPAP/BPAP Devices',
      formula:'=COUNTIF(Stock_Device!E:E,"Active")',
      bg:'#dce8f7',
      fg:'#1a4fa0'
    },
    {
      col:'D',
      label:'Masks',
      formula:'=SUMIF(Stock_Mask!E:E,"Active",Stock_Mask!M:M)',
      bg:'#d6f0e3',
      fg:'#1a7a4a'
    },
    {
      col:'F',
      label:'Accessories',
      formula:'=SUMIF(Stock_Accessory!E:E,"Active",Stock_Accessory!J:J)',
      bg:'#fdf3d0',
      fg:'#a06a00'
    },
    {
      col:'H',
      label:'Total In Stock',
      formula:'=COUNTIF(Stock_Device!E:E,"Active")+SUMIF(Stock_Mask!E:E,"Active",Stock_Mask!M:M)+SUMIF(Stock_Accessory!E:E,"Active",Stock_Accessory!J:J)',
      bg:'#ece5f7',
      fg:'#4a3580'
    }
  ];

  cards.forEach(c => {
    ws.getRange(c.col+'4').setValue(c.label)
      .setBackground(c.bg).setFontFamily('Arial').setFontSize(10).setFontWeight('bold').setFontColor(c.fg)
      .setHorizontalAlignment('center').setVerticalAlignment('middle')
      .setBorder(true,true,false,true,false,false,'#d0d0d0',SpreadsheetApp.BorderStyle.SOLID_MEDIUM);

    ws.getRange(c.col+'5').setBackground(c.bg)
      .setBorder(false,true,false,true,false,false,'#d0d0d0',SpreadsheetApp.BorderStyle.SOLID_MEDIUM);

    ws.getRange(c.col+'6').setFormula(c.formula)
      .setBackground(c.bg).setFontFamily('Arial').setFontSize(24).setFontWeight('bold').setFontColor(c.fg)
      .setHorizontalAlignment('center').setVerticalAlignment('middle')
      .setBorder(false,true,false,true,false,false,'#d0d0d0',SpreadsheetApp.BorderStyle.SOLID_MEDIUM);

    ws.getRange(c.col+'7').setBackground(c.bg)
      .setBorder(false,true,false,true,false,false,'#d0d0d0',SpreadsheetApp.BorderStyle.SOLID_MEDIUM);

    ws.getRange(c.col+'8').setValue('items in stock')
      .setBackground(c.bg).setFontFamily('Arial').setFontSize(9).setFontColor('#888888')
      .setHorizontalAlignment('center').setVerticalAlignment('middle')
      .setBorder(false,true,true,true,false,false,'#d0d0d0',SpreadsheetApp.BorderStyle.SOLID_MEDIUM);
  });

  const colW = {A:2,B:28,C:2,D:28,E:2,F:28,G:2,H:28,I:2};
  Object.keys(colW).forEach(c => {
    ws.setColumnWidth(ws.getRange(c+'1').getColumn(), colW[c] * 7);
  });

  buildDashboardSummaryTables(ws);

  return ws;
}

function buildDashboardSummaryTables(ws) {
  const sections = [
    {
      title: 'Device Stock by Model',
      titleRange: 'A11:D11',
      headerRange: 'A12:D12',
      formulaCell: 'A12',
      color: '#1a4fa0',
      formula: "=QUERY(Stock_Device!D3:G,\"select G,D,F,count(D) where E = 'Active' group by G,D,F label G 'Model', D 'Product Name', F 'Brand', count(D) 'Qty'\",0)"
    },
    {
      title: 'Mask Stock by Model / Size',
      titleRange: 'F11:K11',
      headerRange: 'F12:K12',
      formulaCell: 'F12',
      color: '#1a7a4a',
      formula: "=QUERY(Stock_Mask!D3:M,\"select G,H,I,D,F,sum(M) where E = 'Active' group by G,H,I,D,F label G 'Model', H 'Size', I 'REF', D 'Product Name', F 'Brand', sum(M) 'Qty'\",0)"
    },
    {
      title: 'Accessory Stock by Product',
      titleRange: 'A30:D30',
      headerRange: 'A31:D31',
      formulaCell: 'A31',
      color: '#a06a00',
      formula: "=QUERY(Stock_Accessory!D3:J,\"select D,F,sum(J) where E = 'Active' group by D,F label D 'Product Name', F 'Brand', sum(J) 'Qty'\",0)"
    },
  ];

  ws.getRange('A10:K48').clearContent().clearFormat();

  sections.forEach(s => {
    ws.getRange(s.titleRange).merge()
      .setValue(s.title)
      .setBackground(s.color)
      .setFontFamily('Arial').setFontSize(11).setFontWeight('bold').setFontColor('#FFFFFF')

    ws.getRange(s.formulaCell).setFormula(s.formula);
    ws.getRange(s.headerRange)

      .setHorizontalAlignment('center')
      .setBorder(true,true,true,true,true,true,'#d0d0d0',SpreadsheetApp.BorderStyle.SOLID);
  });

  ws.getRange('A12:K48')
    .setFontFamily('Arial')
    .setFontSize(9)
    .setVerticalAlignment('middle');
  ws.getRange('D12:D48').setHorizontalAlignment('center');
  ws.getRange('K12:K48').setHorizontalAlignment('center');
  ws.getRange('D31:D48').setHorizontalAlignment('center');

  [11, 30].forEach(r => ws.setRowHeight(r, 24));
  ws.setColumnWidth(1, 130);  // A
  ws.setColumnWidth(2, 210);  // B
  ws.setColumnWidth(3, 130);  // C
  ws.setColumnWidth(4, 60);   // D
  ws.setColumnWidth(6, 120);  // F
  ws.setColumnWidth(7, 55);   // G
  ws.setColumnWidth(8, 80);   // H
  ws.setColumnWidth(9, 210);  // I
  ws.setColumnWidth(10, 130); // J
  ws.setColumnWidth(11, 60);  // K
}

function rebuildDashboard() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let ws = ss.getSheetByName(TABS.dashboard);
  if (!ws) ws = ss.insertSheet(TABS.dashboard, 0);
  buildDashboard(ss, ws);
  ss.setActiveSheet(ws);
}

function syncProductDB() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(TABS.productdb);
  if (!sheet) sheet = ss.insertSheet(TABS.productdb);

  sheet.getRange('A1:M1').merge().setValue('Product Database - Skyline Medical')
    .setBackground('#333333').setFontFamily('Arial').setFontSize(13).setFontWeight('bold').setFontColor('#FFFFFF')
    .setHorizontalAlignment('center').setVerticalAlignment('middle');
  styleHeaderRow(sheet, 2, PRODUCTDB_HEADERS, '#333333');

  const existingRows = sheet.getLastRow() >= 3
    ? sheet.getRange(3, 1, sheet.getLastRow() - 2, Math.max(sheet.getLastColumn(), PRODUCTDB_HEADERS.length)).getValues()
    : [];
  const legacyRows = existingRows.filter(r => String(r[4] || '').trim()).map(r => normalizeProductDBRow_(r));
  const seedRows = getLegacyProductDBSeed_().map(r => normalizeProductDBRow_(r));
  const masterRows = SKYLINE_PRODUCT_MASTER.map(r => normalizeProductDBRow_(r));
  const combined = dedupeProductRows_(seedRows.concat(masterRows).concat(legacyRows));

  if (sheet.getLastRow() > 2) {
    sheet.getRange(3, 1, sheet.getLastRow() - 2, Math.max(sheet.getLastColumn(), PRODUCTDB_HEADERS.length)).clearContent();
  }
  if (combined.length) {
    sheet.getRange(3, 1, combined.length, PRODUCTDB_HEADERS.length).setValues(combined);
    combined.forEach((_, i) => {
      const r = i + 3;
      sheet.getRange(r, 1, 1, PRODUCTDB_HEADERS.length)
        .setBackground(r % 2 === 0 ? '#f7f8fa' : '#FFFFFF')
        .setFontFamily('Arial').setFontSize(10)
        .setBorder(true,true,true,true,true,true,'#d0d0d0',SpreadsheetApp.BorderStyle.SOLID);
    });
  }
  SpreadsheetApp.flush();
  Logger.log('ProductDB synced: ' + combined.length + ' rows (' + masterRows.length + ' master SKU)');
}

function normalizeProductDBRow_(r) {
  const row = (r || []).slice(0, PRODUCTDB_HEADERS.length);
  while (row.length < PRODUCTDB_HEADERS.length) row.push('');
  row[6] = normalizeType_(row[6], row[11], row[4]);
  row[12] = toBool_(row[12]) || hasBarcodePattern_(row[0]) || hasBarcodePattern_(row[1]) || hasBarcodePattern_(row[2]) || hasBarcodePattern_(row[3]);
  return row;
}

function dedupeProductRows_(rows) {
  const out = [];
  const seen = new Set();
  rows.forEach(r => {
    if (!String(r[4] || '').trim()) return;
    const scanKey = [r[0], r[1], r[2], r[3]].join('|').toUpperCase();
    const manualKey = ['MANUAL', r[10], r[4], r[5], r[8]].join('|').toUpperCase();
    const key = scanKey.replace(/\|/g, '') ? scanKey : manualKey;
    if (seen.has(key)) return;
    seen.add(key);
    out.push(r);
  });
  return out;
}

function learnProductMapping_(ss, b) {
  const m = b && b.learnMapping ? b.learnMapping : null;
  if (!m) return false;

  const productCode = String(m.product_code || m.productCode || b.productCode || '').trim();
  const gtin = String(m.gtin || '').replace(/[^\x20-\x7E]/g, '').replace(/\s/g, '').replace(/^0+/, '').toUpperCase();
  const sn = String(m.sn || '').replace(/[^\x20-\x7E]/g, '').replace(/\s/g, '').toUpperCase();
  const lot = normalizeLot(String(m.lot || ''));
  if (!productCode || (!gtin && !sn && !lot)) return false;

  const sheet = buildProductDBSheet(ss);
  const last = sheet.getLastRow();
  const width = Math.max(sheet.getLastColumn(), PRODUCTDB_HEADERS.length);
  const rows = last >= 3
    ? sheet.getRange(3, 1, last - 2, width).getValues().map(r => normalizeProductDBRow_(r))
    : [];

  const source = rows.find(r => String(r[10] || '').trim() === productCode);
  if (!source) {
    Logger.log('Learn ProductDB mapping skipped: product_code not found: ' + productCode);
    return false;
  }

  const learned = normalizeProductDBRow_([
    gtin,
    sn,
    '',
    lot,
    source[4],
    source[5],
    source[6],
    source[7],
    source[8],
    source[9],
    source[10],
    source[11],
    true
  ]);

  const combined = dedupeProductRows_(rows.concat([learned]));
  if (combined.length === rows.length) return false;

  if (last > 2) {
    sheet.getRange(3, 1, last - 2, width).clearContent();
  }
  sheet.getRange(3, 1, combined.length, PRODUCTDB_HEADERS.length).setValues(combined);
  combined.forEach((_, i) => {
    const r = i + 3;
    sheet.getRange(r, 1, 1, PRODUCTDB_HEADERS.length)
      .setBackground(r % 2 === 0 ? '#f7f8fa' : '#FFFFFF')
      .setFontFamily('Arial').setFontSize(10)
      .setBorder(true,true,true,true,true,true,'#d0d0d0',SpreadsheetApp.BorderStyle.SOLID);
  });
  SpreadsheetApp.flush();
  Logger.log('Learn ProductDB mapping added for ' + productCode);
  return true;
}

function normalizeType_(type, category, name) {
  const t = String(type || '').trim().toLowerCase();
  if (['device','mask','accessory'].indexOf(t) !== -1) return t;
  const c = String(category || '');
  const n = String(name || '').toLowerCase();
  const m = c.match(/^\s*(\d+)/);
  const catNo = m ? parseInt(m[1], 10) : 0;
  if ([5,6,7].indexOf(catNo) !== -1) return 'mask';
  if (/filter|hose|circuit|cushion|humidifier|chamber|accessor|chin strap/i.test(c + ' ' + n)) return 'accessory';
  if ([1,2,3,4].indexOf(catNo) !== -1) return 'device';
  return 'accessory';
}

function toBool_(v) {
  if (v === true) return true;
  const s = String(v || '').trim().toUpperCase();
  return s === 'TRUE' || s === 'YES' || s === '1';
}

function hasBarcodePattern_(v) {
  const s = String(v || '').trim().toUpperCase();
  if (!s || s === '-') return false;
  return /^\d{8,14}$/.test(s) || /(BY-|R30F|R25V|R20I|R20A|NM\d|PMI|HP\d|SPO2|\*)/.test(s);
}

function readProductDB(ss) {
  const sheet = ss.getSheetByName(TABS.productdb);
  if (!sheet) return [];
  const last = sheet.getLastRow();
  if (last < 3) return [];

  const headers = sheet.getRange(2, 1, 1, Math.max(sheet.getLastColumn(), PRODUCTDB_HEADERS.length)).getValues()[0];
  const idx = {};
  headers.forEach((h, i) => idx[String(h || '').trim()] = i);
  const col = name => idx[name] == null ? PRODUCTDB_HEADERS.indexOf(name) : idx[name];
  const data = sheet.getRange(3, 1, last - 2, Math.max(sheet.getLastColumn(), PRODUCTDB_HEADERS.length)).getValues();
  return data
    .filter(r => r[col('Product Name')])
    .map(r => ({
      gtin:         String(r[col('GTIN')] || '').trim(),
      sn_exact:     String(r[col('SN_Exact')] || '').trim().toUpperCase(),
      sn_pfx:       String(r[col('SN_Prefix')] || '').trim().toUpperCase(),
      lot_pfx:      String(r[col('LOT_Prefix')] || '').trim().toUpperCase(),
      name:         String(r[col('Product Name')] || '').trim(),
      brand:        String(r[col('Brand')] || '').trim(),
      type:         normalizeType_(r[col('Type')], r[col('Category')], r[col('Product Name')]),
      model:        String(r[col('Model')] || '').trim(),
      size:         String(r[col('Size')] || '').trim(),
      ref:          String(r[col('REF')] || '').trim(),
      product_code: String(r[col('Product Code')] || '').trim(),
      category:     String(r[col('Category')] || '').trim(),
      has_barcode:  toBool_(r[col('has_barcode')])
    }));
}

function backfillDeviceSheet_(sheet, products) {
  if (!sheet || sheet.getLastRow() < DATA_START) return 0;
  const values = sheet.getRange(DATA_START, 1, sheet.getLastRow() - DATA_START + 1, sheet.getLastColumn()).getValues();
  let updated = 0;
  values.forEach((row, i) => {
    const p = lookupProductForBackfill_(products, {
      type:'device',
      gtin: row[9],
      lot: row[DEV_COL_LOT - 1],
      sn: row[DEV_COL_SN - 1]
    });
    if (!p || p.type !== 'device') return;
    const r = DATA_START + i;
    updated += setIfChanged_(sheet, r, 4, p.name);
    updated += setIfChanged_(sheet, r, 6, p.brand);
    updated += setIfChanged_(sheet, r, 7, p.model);
  });
  return updated;
}

}

  if (!sheet || sheet.getLastRow() < DATA_START) return 0;

function setIfChanged_(sheet, row, col, value) {
  value = String(value || '').trim();
  if (!value) return 0;
  const cur = String(sheet.getRange(row, col).getValue() || '').trim();
  if (cur === value) return 0;
  sheet.getRange(row, col).setValue(value);
  return 1;
}

function lookupProductForBackfill_(products, d) {
  const gtin = String(d.gtin || '').replace(/\s/g, '').replace(/^0+/, '').toUpperCase();
  const lot = normalizeLot(d.lot);
  const sn = String(d.sn || '').replace(/\s/g, '').toUpperCase();
  let wildcardLotHit = null;

  for (const p of products || []) {
    if (p.type && d.type && p.type !== d.type) continue;
    const pGtin = String(p.gtin || '').replace(/^0+/, '').toUpperCase();
    if (pGtin && gtin && pGtin === gtin) return mergeLotDetails_(p, lot);
    if (p.sn_exact && sn && p.sn_exact === sn) return mergeLotDetails_(p, lot);
    if (p.sn_pfx && sn && sn.startsWith(p.sn_pfx)) return mergeLotDetails_(p, lot);
    if (p.lot_pfx && lot && matchProductPattern_(lot, p.lot_pfx)) {
      if (String(p.lot_pfx).indexOf('*') !== -1) {
        if (!wildcardLotHit) wildcardLotHit = p;
      } else {
        return mergeLotDetails_(p, lot);
      }
    }
  }

  if (wildcardLotHit) return mergeLotDetails_(wildcardLotHit, lot);
  return inferProductFromLot_(lot);
}

function matchProductPattern_(value, pattern) {
  const v = String(value || '').trim().toUpperCase();
  const p = String(pattern || '').trim().toUpperCase();
  if (!v || !p) return false;
  if (p.indexOf('*') === -1) return v.indexOf(p) === 0;
  const re = new RegExp('^' + p.split('*').map(escapeRegExp_).join('.*') + '$');
  return re.test(v);
}

function escapeRegExp_(s) {
  return String(s || '').replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}

function mergeLotDetails_(product, lot) {
  const inferred = inferProductFromLot_(lot) || {};
  const merged = {
    name: product.name || inferred.name || '',
    brand: product.brand || inferred.brand || '',
    type: product.type || inferred.type || '',
    model: product.model || inferred.model || '',
    size: product.size || inferred.size || '',
    ref: product.ref || inferred.ref || '',
  };
  if (inferred.size) merged.size = inferred.size;
  if (inferred.ref) merged.ref = inferred.ref;
  return merged;
}

function inferProductFromLot_(lot) {
  lot = normalizeLot(lot);
  if (!lot) return null;
  const rules = [
    { test: l => l.indexOf('NM6A') !== -1, type:'mask', model:'EaseFit NM6A', name:'Beyond EaseFit NM6A', brand:'Hunan Beyond' },
    { test: l => l.indexOf('NM6') !== -1,  type:'mask', model:'EaseFit NM6',  name:'Beyond EaseFit NM6',  brand:'Hunan Beyond' },
    { test: l => l.indexOf('NM5A') !== -1, type:'mask', model:'EaseFit NM5A', name:'Beyond EaseFit NM5A', brand:'Hunan Beyond' },
    { test: l => l.indexOf('NM5') !== -1,  type:'mask', model:'EaseFit NM5',  name:'Beyond EaseFit NM5',  brand:'Hunan Beyond' },

function doGet(e) {
  const p = (e && e.parameter) ? e.parameter : {};
  const action = p.action || '';

  if (action === 'login') {
  const user = p.user || '';
  const pass = p.pass || '';
  const staffName = String(p.staffName || '').trim();
  if (USERS[user] && USERS[user] === pass && staffName) {
    return ok({ message:'ok', user:user, role:getUserRole(user), staffName:staffName, token:makeAuthToken(user, staffName) });
  }
  return err('ชื่อผู้ใช้ รหัสผ่าน หรือชื่อพนักงานไม่ถูกต้อง');
  }
  if (action === 'ping') {
    return ok({message:'Skyline API ready', ts:new Date().toISOString()});
  }

  if (action === 'stock') {
    try {
      const auth = requireAuth(p);
      if (!auth.ok) return err(auth.error);
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      return ok({
        device: readStock(ss, 'device'),
        mask: readStock(ss, 'mask'),
        accessory: readStock(ss, 'accessory')
      });
    } catch(ex) {
      return err(ex.toString());
    }
  }

  if (action === 'history') {
    try {
      const auth = requireAuth(p);
      if (!auth.ok) return err(auth.error);
      const n      = parseInt(p.n || '100', 10);
      const offset = parseInt(p.offset || '0', 10);
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      return ok({history: readHistory(ss, n, offset)});
    } catch(ex) {
      return err(ex.toString());
    }
  }

  if (action === 'products') {
    try {
      const auth = requireAuth(p);
      if (!auth.ok) return err(auth.error);
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      return ok({products: readProductDB(ss)});
    } catch(ex) {
      return err(ex.toString());
    }
  }

  try {
    return HtmlService.createHtmlOutputFromFile('index')
      .setTitle('Skyline Stock')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  } catch(ex) {
    return ok({message:'Skyline API ready - HTML not found', ts:new Date().toISOString()});
  }
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    const ss   = SpreadsheetApp.getActiveSpreadsheet();
    const now  = new Date();
    const date = Utilities.formatDate(now, TZ, 'dd/MM/yyyy');
    const time = Utilities.formatDate(now, TZ, 'HH:mm');

    // ?? Login via POST (鉊?葦鉊晤葵鉊?鉊耜?鉆腹鉆?鉊艇鉆?鉊?URL) ????????????????
    if (body.action === 'login') {
      const user = String(body.user || '');
      const pass = String(body.pass || '');
      const staffName = String(body.staffName || '').trim();
      if (USERS[user] && USERS[user] === pass && staffName) {
        return ok({ message:'ok', user:user, role:getUserRole(user), staffName:staffName, token:makeAuthToken(user, staffName) });
      }
      return err('ชื่อผู้ใช้ รหัสผ่าน หรือชื่อพนักงานไม่ถูกต้อง');
    }

    // ?? Validate type ??????????????????????????????????????????
    const VALID_TYPES = ['device', 'mask', 'accessory'];
    if (['in', 'out', 'check', 'out_by_id'].includes(body.action)) {
      if (!VALID_TYPES.includes(body.type)) {
        return err('ประเภทสินค้าไม่ถูกต้อง: ' + String(body.type || ''));
      }
    }

    const auth = requireAuth(body);
    if (!auth.ok) return err(auth.error);
    applyAuthContext_(body, auth);

    if (body.action === 'in') {
      try { validateTransaction_(body, 'in'); } catch(vErr) { return err(vErr.message); }
      const duplicate = findDuplicate(ss, body.type, body.sn, body.lot, body.name, body.size || '');
      if (duplicate) return err('DUPLICATE', { duplicate: duplicate });

      let stockInResult;
      try {
        stockInResult = doStockIn(ss, body, date, time);
      } catch(stockErr) {
        Logger.log('Stock In error: ' + stockErr.toString());
        return err('Stock In failed: ' + (stockErr.message || stockErr.toString()));
      }
      try {
        learnProductMapping_(ss, body);
      } catch(learnErr) {
        Logger.log('Learn ProductDB mapping failed after Stock In: ' + (learnErr.message || learnErr.toString()));
      }

      return ok(Object.assign({message:'in'}, typeof stockInResult === 'object' ? stockInResult : {row: stockInResult}));
    }

    if (body.action === 'out') {
      try { validateTransaction_(body, 'out'); } catch(vErr) { return err(vErr.message); }
      let r;
      try {
        r = doStockOut(ss, body, date, time);
      } catch(stockErr) {
        Logger.log('Stock Out error: ' + stockErr.toString());
        return err('Stock Out failed: ' + (stockErr.message || stockErr.toString()));
      }
      if (r !== -1) {
        try {
          learnProductMapping_(ss, body);
        } catch(learnErr) {
          Logger.log('Learn ProductDB mapping failed after Stock Out: ' + (learnErr.message || learnErr.toString()));
        }
      }
      return r === -1 ? err('Item not found') : ok({message:'out', row:r});
    }

    if (body.action === 'check') {
      const exists = checkDuplicate(ss, body.type, body.sn, body.lot, body.name, body.size || '');
      return ok({ exists });
    }

    if (body.action === 'out_by_id') {
      try { validateTransaction_(body, 'out'); } catch(vErr) { return err(vErr.message); }
      try {
        doStockOutByRow(ss, body, date, time);
      } catch(ex) {
        Logger.log('out_by_id error: ' + ex.toString());
        return err('out_by_id failed: ' + (ex.message || ex.toString()));
      }
      return ok({message:'out_by_id'});
    }

    if (body.action === 'batch_stocktake') {
      try {
        const result = doBatchStocktake(ss, body.items || [], date, time, body);
        return ok({message:'batch_stocktake', updated:result.updated});
      } catch(ex) {
        Logger.log('batch_stocktake error: ' + ex.toString());
        return err('Batch stocktake failed: ' + (ex.message || ex.toString()));
      }
    }

    if (body.action === 'sort') {
      const admin = requireAuth(body, 'admin');
      if (!admin.ok) return err(admin.error);
      try {
        const sheet = ss.getSheetByName(TABS.device);
        if (!sheet) return err('ไม่พบ Device Sheet');
        sortDeviceSheet(sheet);
        return ok({message:'sorted'});
      } catch(ex) {
        Logger.log('sort error: ' + ex.toString());
        return err('sort failed: ' + (ex.message || ex.toString()));
      }
    }

    return err('Unknown action');
  } catch(ex) {
    Logger.log('doPost fatal error: ' + ex.toString());
    return err('เกิดข้อผิดพลาดในระบบ กรุณาลองใหม่อีกครั้ง');
  }
}

function readStock(ss, type) {
  const sheet = ss.getSheetByName(TABS[type]);
  if (!sheet) return [];
  const last = sheet.getLastRow();
  if (last < DATA_START) return [];

  const headers = sheet.getRange(HDR_ROW,1,1,sheet.getLastColumn()).getValues()[0];
  const data = sheet.getRange(DATA_START,1,last-DATA_START+1,sheet.getLastColumn()).getValues();

  return data.map((row, i) => {
    const o = {_sheetRow: i + DATA_START};
    headers.forEach((h, j) => {
      if (h) o[h] = row[j];
    });
    return o;
  }).filter(r => r['Product Name'] && r['Status'] === 'Active');
}

function readHistory(ss, n, offset) {
  const sheet = ss.getSheetByName(TABS.history);
  if (!sheet) return [];
  ensureHistoryColumns_(sheet);
  const last = sheet.getLastRow();
  if (last < DATA_START) return [];

  const headers = sheet.getRange(HDR_ROW, 1, 1, sheet.getLastColumn()).getValues()[0];
  const totalRows = last - DATA_START + 1;

  // offset 鉊萵鉊?鉊耜?鉆?鉊抉艇鉆葡鉊芹虜鉊?(鉊腦鉊啤葷鉊晤?鉊毯?鉊徇腹鉆葵鉊詮?)
  const off = parseInt(offset || 0, 10);
  if (off >= totalRows) return [];  // 鉆腹鉆腹鉊菽?鉆葉鉊﹤號鉊丞?鉊葩鉆腹鉆艇鉆葷
  const startFromBottom = Math.max(DATA_START, last - off - n + 1);
  const endRow = Math.max(DATA_START, last - off);
  const count = endRow - startFromBottom + 1;
  if (count <= 0) return [];

  const data = sheet.getRange(startFromBottom, 1, count, sheet.getLastColumn()).getValues();

  return data.reverse().map(row => {
    const o = {};
    headers.forEach((h, j) => { if (h) o[h] = row[j]; });
    return normalizeHistoryRecord_(o);
  }).filter(r => r['Product Name']);
}

function ensureHistoryColumns_(sheet) {
  if (!sheet) return;
  const expected = [
    'Date','Time','Type','Product Name','Brand','Serial / LOT','Action','Qty',
    'Before Qty','After Qty','Purpose','Location / Recipient','Recorded By','Operator','Note','Reference'
  ];
  if (sheet.getMaxColumns() < expected.length) {
    sheet.insertColumnsAfter(sheet.getMaxColumns(), expected.length - sheet.getMaxColumns());
  }
  const width = Math.max(sheet.getLastColumn(), expected.length);
  const headers = sheet.getRange(HDR_ROW, 1, 1, width).getValues()[0];
  let changed = false;
  expected.forEach((name, i) => {
    const cur = String(headers[i] || '').trim();
    if (!cur || (i === 11 && cur === 'Recipient')) {
      headers[i] = name;
      changed = true;
    }
  });
  if (changed) sheet.getRange(HDR_ROW, 1, 1, width).setValues([headers]);
}

function normalizeHistoryRecord_(o) {
  if (!o) return {};
  if (!o['Location / Recipient']) {
    o['Location / Recipient'] = o['Recipient'] || o['Location'] || o['Location / Hospital'] || '';
  }
  if (!o['Recorded By']) {
    o['Recorded By'] = o['Record by'] || o['Record By'] || o['record by'] || o['User'] || '';
  }
  if (!o['Purpose']) {
    o['Purpose'] = o['วัตถุประสงค์'] || '';
  }
  return o;
}

function normalizeText(v) {
  return String(v || '').trim();
}

function getUserRole(user) {
  return ADMIN_USERS[user] ? 'admin' : 'stock';
}

function getAuthSecret() {
  const props = PropertiesService.getScriptProperties();
  let secret = props.getProperty('SKL_AUTH_SECRET');
  if (!secret) {
    secret = Utilities.getUuid() + ':' + Utilities.getUuid();
    props.setProperty('SKL_AUTH_SECRET', secret);
  }
  return secret;
}

function makeAuthToken(user, staffName) {
  const secret = getAuthSecret();
  const subject = String(user) + '|' + String(staffName || '').trim();
  const sig = Utilities.computeHmacSha256Signature(subject, secret);
  return Utilities.base64EncodeWebSafe(sig);
}

function requireAuth(payload, requiredRole) {
  const user = String(payload.user || '');
  const staffName = String(payload.staffName || payload.operator || '').trim();
  const token = String(payload.token || '');
  if (!USERS[user] || !staffName || !token || token !== makeAuthToken(user, staffName)) {
    return { ok:false, error:'Unauthorized or missing staff name' };
  }
  if (requiredRole === 'admin' && getUserRole(user) !== 'admin') {
    return { ok:false, error:'Admin permission required' };
  }
  return { ok:true, user:user, role:getUserRole(user), staffName:staffName };
}

function normalizeLot(lot) {
  lot = String(lot || '').trim().toUpperCase();
  if (!lot) return '';

  const idxBy = lot.indexOf('BY-');
  if (idxBy !== -1) return lot.substring(idxBy);

  // 鉊萵鉊?鉊Ｒ萼鉊萵鉊抉?鉊丞? GS1 鉊?鉊耜?鉊徇?鉆葡鉊冢葉鉊?鉆艇鉊啤?鉊晤? C.\d+ patterns (sync 鉊萵鉊?frontend)
  return lot.replace(/^\d{10,}/, '').replace(/C\.\d+(?:\.\d+)+/g, '').trim();
}

function findExistingAggregateRow(sheet, type, b) {
  const last = sheet.getLastRow();
  if (last < DATA_START) return -1;

  const numCols = sheet.getLastColumn();
  const data = sheet.getRange(DATA_START, 1, last - DATA_START + 1, numCols).getValues();

  const lotCol  = type === 'mask' ? MASK_COL_LOT  : ACC_COL_LOT;
  const sizeCol = MASK_COL_SIZE;

  for (let i = 0; i < data.length; i++) {
    const row = data[i];
    if (normalizeText(row[COL_STATUS - 1]) !== 'Active') continue;

    const productName = normalizeText(row[3]);   // col 4
    const brand       = normalizeText(row[5]);   // col 6
    const lot         = normalizeText(row[lotCol - 1]);
    const size        = type === 'mask' ? normalizeText(row[sizeCol - 1]) : '';

    const sameName  = productName === normalizeText(b.name);
    const sameBrand = brand       === normalizeText(b.brand);
    const sameLot   = normalizeLot(lot) === normalizeLot(b.lot);
    const sameSize  = (type !== 'mask') || (size === normalizeText(b.size));

    if (sameName && sameBrand && sameLot && sameSize) {
      return i + DATA_START;  // 鉊虞鉊?row number 鉊腦鉊毯?鉆? sheet
    }
  }

  return -1;
}

function rowMatchesPayload(row, type, b) {
  if (normalizeText(row[COL_STATUS - 1]) !== 'Active') return false;

  const rowName = normalizeText(row[3]).toUpperCase();
  const rowBrand = normalizeText(row[5]).toUpperCase();
  if (b.name && rowName !== normalizeText(b.name).toUpperCase()) return false;
  if (b.brand && rowBrand !== normalizeText(b.brand).toUpperCase()) return false;

  if (type === 'device') {
    const rowSn = normalizeText(row[DEV_COL_SN - 1]).toUpperCase();
    const rowLot = normalizeLot(row[DEV_COL_LOT - 1]);
    const rowGtin = normalizeText(row[9]).replace(/\s/g, '').replace(/^0+/, '').toUpperCase(); // col 10
    const sn = normalizeText(b.sn).toUpperCase();
    const lot = normalizeLot(b.lot);
    const gtin = normalizeText(b.gtin).replace(/\s/g, '').replace(/^0+/, '').toUpperCase();
    if (sn) return rowSn === sn;
    if (lot) return rowLot === lot;
    if (gtin) return rowGtin === gtin;
    return false;
  }

  const lotCol = type === 'mask' ? MASK_COL_LOT : ACC_COL_LOT;
  const gtinCol = type === 'mask' ? MASK_COL_GTIN : ACC_COL_GTIN;
  const rowLot = normalizeLot(row[lotCol - 1]);
  const rowGtin = normalizeText(row[gtinCol - 1]).replace(/\s/g, '').replace(/^0+/, '').toUpperCase();
  const lot = normalizeLot(b.lot);
  const gtin = normalizeText(b.gtin).replace(/\s/g, '').replace(/^0+/, '').toUpperCase();
  if (lot && rowLot !== lot) return false;
  if (!lot && gtin && rowGtin !== gtin) return false;
  if (!lot && !gtin) return false;

  if (type === 'mask' && b.size) {
    const rowSize = normalizeText(row[MASK_COL_SIZE - 1]).toUpperCase();
    if (rowSize !== normalizeText(b.size).toUpperCase()) return false;
  }
  return true;
}

function validateTransaction_(b, action) {
  if (!b) throw new Error('Missing payload');
  if (!b.type || ['device','mask','accessory'].indexOf(String(b.type)) === -1) throw new Error('Invalid product type');
  if (!String(b.name || '').trim()) throw new Error('Missing product name');
  const qty = parseInt(b.qty || 1, 10);
  if (!qty || qty <= 0) throw new Error('Quantity must be greater than 0');
  if (!String(b.operator || b.staffName || b.user || '').trim()) throw new Error('Missing operator');
  if (b.type === 'device' && !String(b.sn || '').trim()) throw new Error('กรุณากรอกเลขเครื่อง (SN)');
  if ((b.type === 'mask' || b.type === 'accessory') && !String(b.lot || '').trim()) throw new Error('กรุณากรอกเลขล็อต (LOT)');
  if (action === 'out') {
    const purpose = String(b.purpose || '').trim();
    const recipient = String(b.recipient || b.location || '').trim();
    if (!purpose) throw new Error('Missing stock-out purpose');
    if (purpose === '\u0e2d\u0e37\u0e48\u0e19\u0e46' && !String(b.purposeOther || '').trim()) throw new Error('Please specify other purpose');
    if (!recipient) throw new Error('Missing recipient');
    b.purpose = purpose === '\u0e2d\u0e37\u0e48\u0e19\u0e46' ? '\u0e2d\u0e37\u0e48\u0e19\u0e46: ' + String(b.purposeOther || '').trim() : purpose;
    b.recipient = recipient;
  }
}

function applyAuthContext_(body, auth) {
  body.user = auth.user;
  body.operator = String(body.staffName || body.operator || auth.staffName || auth.user || '').trim();
  return body;
}

 * ========================================================= */
function doStockIn(ss, b, date, time) {
  const sheet = ss.getSheetByName(TABS[b.type]);
  if (!sheet) throw new Error('Sheet not found: ' + b.type);

  b.lot = normalizeLot(b.lot);

  // DEVICE = 鉊?葡鉊Ｒ?鉊腦鉊獅?鉊冢?
  if (b.type === 'device') {
    const seq = sheet.getLastRow() - HDR_ROW + 1;
    const row = [
      seq,
      date,
      time,
      b.name || '',
      'Active',
      b.brand || '',
      b.model || '',
      b.sn || '',
      b.lot || '',
      b.gtin || '',
      b.mfgDate || '',
      b.expiry || '',
      b.location || '',
      '',
      b.user || '',
      b.note || ''
    ];

    sheet.appendRow(row);
    const newRow = sheet.getLastRow();
    styleDataRow(sheet, newRow, b.type);
    appendHistory(ss, b, date, time, 'Stock In');
    return newRow;

    const newQty = oldQty + addQty;
    sheet.getRange(existingRow, qtyCol).setValue(newQty);

    if (b.note) {
      const noteCol = (b.type === 'mask') ? MASK_COL_NOTE : ACC_COL_NOTE;
      const oldNote = normalizeText(sheet.getRange(existingRow, noteCol).getValue());
      sheet.getRange(existingRow, noteCol).setValue(oldNote ? oldNote + ' | ' + b.note : b.note);
    }

    if (normalizeText(sheet.getRange(existingRow, COL_STATUS).getValue()) !== 'Active') {
      sheet.getRange(existingRow, COL_STATUS).setValue('Active');
    }

    sheet.getRange(existingRow,1,1,sheet.getLastColumn()).setBackground('#FFFFFF');
    styleDataRow(sheet, existingRow, b.type);
    if (_needsSort(sheet, COL_STATUS, 2)) sortMaskAccSheet(sheet);
    b.beforeQty = oldQty;
    b.afterQty = newQty;
    appendHistory(ss, b, date, time, 'Stock In');
    return {row: existingRow, merged: true, newQty: newQty};
  }

  const seq = sheet.getLastRow() - HDR_ROW + 1;
  let row = [];

  if (b.type === 'mask') {
    row = [
      seq,
      date,
      time,
      b.name || '',
      'Active',
      b.brand || '',
      b.model || '',
      b.size || '',
      b.ref || '',
      b.lot || '',
      b.gtin || '',
      b.mfgDate || '',
      addQty,
      b.note || '',
      b.expiry || ''
    ];
  } else if (b.type === 'accessory') {
    row = [
      seq,
      date,
      time,
      b.name || '',
      'Active',
      b.brand || '',
      b.lot || '',
      b.gtin || '',
      b.mfgDate || '',
      addQty,
      b.location || '',
      b.note || ''
    ];
  }

  sheet.appendRow(row);
  const newRow = sheet.getLastRow();
  styleDataRow(sheet, newRow, b.type);
  if (_needsSort(sheet, COL_STATUS, 2)) sortMaskAccSheet(sheet);
  appendHistory(ss, b, date, time, 'Stock In');
  return newRow;
}

/* =========================================================
 * STOCK OUT
 * ========================================================= */
function doStockOut(ss, b, date, time) {
  const sheet = ss.getSheetByName(TABS[b.type]);
  if (!sheet) throw new Error('Sheet not found: ' + b.type);

  b.lot = normalizeLot(b.lot);

  const last = sheet.getLastRow();
  if (last < DATA_START) return -1;

  const numCols = sheet.getLastColumn();
  const data = sheet.getRange(DATA_START, 1, last - DATA_START + 1, numCols).getValues();

  // DEVICE = 鉊冢葉鉊?鉊耜腹 SN 鉆鉊?鉊葦鉊丞萵鉊?(scan 鉊葡鉊艇鉆葡鉊?鉊嗣?鉊?鉊?鉆鉊葉鉊丞?鉊耜葵鉊詮?鉊?鉊冢?)
  if (b.type === 'device') {
    const keyVal = b.sn || b.lot || '';
    const gtin = normalizeText(b.gtin).replace(/\s/g, '').replace(/^0+/, '').toUpperCase();
    for (let i = data.length - 1; i >= 0; i--) {
      const row = data[i];
      if (normalizeText(row[COL_STATUS - 1]) !== 'Active') continue;

      const rowSn  = normalizeText(row[DEV_COL_SN - 1]);
      const rowLot = normalizeText(row[DEV_COL_LOT - 1]);
      const rowGtin = normalizeText(row[9]).replace(/\s/g, '').replace(/^0+/, '').toUpperCase();
      const r = i + DATA_START;

      const snMatch = b.sn && rowSn.toUpperCase() === normalizeText(b.sn).toUpperCase();
      const lotMatch = !b.sn && b.lot && normalizeLot(rowLot) === normalizeLot(keyVal);
      const gtinMatch = !b.sn && !b.lot && gtin && rowGtin === gtin;
      if (snMatch || lotMatch || gtinMatch) {
        markOut(sheet, r, b.type, date);

        if (b.location || b.note) {
          const oldNote = normalizeText(row[15]);  // col 16 = Note
          const extra = (b.location ? ('OUT TO: ' + b.location) : '') + (b.note ? (' | ' + b.note) : '');
          sheet.getRange(r, 16).setValue(oldNote ? oldNote + ' | ' + extra : extra);
        }

        appendHistory(ss, b, date, time, 'Stock Out');
        return r;
      }
    }
    return -1;
  }

  // MASK / ACCESSORY = 鉊丞? Qty 鉆? Active row + 鉊芹腦鉆葡鉊?Out row 鉆葦鉊﹤?鉊?萼鉊虜鉊號鉆腦鉊晤?
  const lotCol  = b.type === 'mask' ? MASK_COL_LOT  : ACC_COL_LOT;
  const qtyCol  = b.type === 'mask' ? MASK_COL_QTY  : ACC_COL_QTY;
  const outQty  = Math.max(1, parseInt(b.qty || 1, 10));

  for (let i = 0; i < data.length; i++) {
    const row = data[i];
    if (normalizeText(row[COL_STATUS - 1]) !== 'Active') continue;

    const rowLot  = normalizeText(row[lotCol - 1]);
    const rowName = normalizeText(row[3]);  // col 4 = Product Name
    const rowSize = b.type === 'mask' ? normalizeText(row[MASK_COL_SIZE - 1]) : '';

    // 鉊腦鉊抉? LOT (鉊萵鉊?鉊晤?)
    if (normalizeLot(rowLot) !== normalizeLot(b.lot)) continue;
    // 鉊腦鉊抉? Product Name (鉊?鉊耜腹鉊?
    if (b.name && rowName.toUpperCase() !== normalizeText(b.name).toUpperCase()) continue;
    // 鉊腦鉊抉? Size 鉊芹董鉊徇腦鉊晤? mask (鉊?鉊耜腹鉊? 鉆鉊虞鉆葉鉆腹鉆?鉆葡鉊Ｒ?鉊毯? size
    if (b.type === 'mask' && b.size && rowSize.toUpperCase() !== normalizeText(b.size).toUpperCase()) continue;

    const r = i + DATA_START;
    const qty = parseInt(row[qtyCol - 1] || 0, 10);
    if (qty <= 0) continue;
    const actualOutQty = Math.min(outQty, qty);

    // 鉊丞? Qty 鉊葡鉊﹤?鉊喪?鉊抉?鉊腦鉊毯?鉊葭鉆?鉆葡鉊Ｒ葉鉊冢? (鉊?鉊耜?鉊徇艇鉊獅葉 0 鉆葦鉆?mark Out)
    if (qty > actualOutQty) {
      sheet.getRange(r, qtyCol).setValue(qty - actualOutQty);
    } else {
      markOut(sheet, r, b.type, date);
    }

    // 鉊芹腦鉆葡鉊?Out row 鉆葦鉊﹤?鉆鉊芹腹鉊?鉆鉊虞鉆葉 track 鉊抉?鉊耜?鉆葡鉊Ｒ?鉊徇?鉆?鉊?
    appendOutRow(sheet, b.type, row, date, time, b.location || '', b.note || '', actualOutQty);
    b.qty = actualOutQty;
    b.beforeQty = qty;
    b.afterQty = Math.max(0, qty - actualOutQty);
    if (qty <= actualOutQty && _needsSort(sheet, COL_STATUS, 2)) sortMaskAccSheet(sheet);

    appendHistory(ss, b, date, time, 'Stock Out');
    return r;
  }

  return -1;
}

/* 鉊芹腦鉆葡鉊?Out row 鉆葦鉊﹤?鉆? Mask / Accessory sheet */
function appendOutRow(sheet, type, srcRow, date, time, location, note, qty) {
  qty = Math.max(1, parseInt(qty || 1, 10));
  const seq = sheet.getLastRow() - HDR_ROW + 1;
  let outRow = [];

  if (type === 'mask') {
    outRow = [
      seq,
      date,
      time,
      srcRow[3],              // Product Name
      'Out',
      srcRow[5],              // Brand
      srcRow[6],              // Model
      srcRow[MASK_COL_SIZE - 1],   // Size
      srcRow[MASK_COL_REF  - 1],   // REF
      srcRow[MASK_COL_LOT  - 1],   // LOT
      srcRow[MASK_COL_GTIN - 1],   // GTIN
      srcRow[MASK_COL_MFG    - 1],   // Mfg Date
      qty,                           // Qty 鉊葭鉆?鉆葡鉊Ｒ葉鉊冢? (鉊葡鉊﹤?鉊?葩鉊?
      (location ? 'OUT TO: ' + location : '') + (note ? (location ? ' | ' : '') + note : ''),
      srcRow[MASK_COL_EXPIRY - 1] || ''  // Expiry
    ];
  } else if (type === 'accessory') {
    outRow = [
      seq,
      date,
      time,
      srcRow[3],              // Product Name
      'Out',
      srcRow[5],              // Brand
      srcRow[ACC_COL_LOT  - 1],    // LOT
      srcRow[ACC_COL_GTIN - 1],    // GTIN
      srcRow[ACC_COL_MFG  - 1],    // Mfg Date
      qty,                    // Qty 鉊葭鉆?鉆葡鉊Ｒ葉鉊冢? (鉊葡鉊﹤?鉊?葩鉊?
      location,               // Location = 鉊虞鉆葉鉊號鉆腦鉊晤?
      note
    ];
  }

  if (!outRow.length) return;

  sheet.appendRow(outRow);
  const newRow = sheet.getLastRow();

  // 鉊芹?鉊艇鉆?鉊葷 Out (鉊芹葭鉆?鉊葉鉆葉鉊?
  sheet.getRange(newRow, 1, 1, sheet.getLastColumn())
    .setBackground('#fde8e8')
    .setFontFamily('Arial')
    .setFontSize(10)
    .setVerticalAlignment('middle')
    .setBorder(true,true,true,true,true,true,'#d0d0d0',SpreadsheetApp.BorderStyle.SOLID);

  sheet.getRange(newRow, COL_STATUS)
    .setFontColor('#b92e2e')
    .setFontWeight('bold');
}

/* 鉆鉊?葭鉊Ｒ? Device sheet: Active 鉊?鉊冢? ??SN A? */
function sortDeviceSheet(sheet) {
  const last = sheet.getLastRow();
  if (last < DATA_START + 1) return;  // 鉊?鉊冢?鉊﹤葭鉊冢腺鉆葡鉊?鉆葉鉊?2 鉆?鉊抉?鉆葉鉊﹤號鉊?
  const range = sheet.getRange(DATA_START, 1, last - DATA_START + 1, sheet.getLastColumn());
  range.sort([
    { column: COL_STATUS,  ascending: true },   // "Active" < "Out" ??Active 鉊葆鉆?鉊?鉊冢?
    { column: DEV_COL_SN,  ascending: true }    // SN A? 鉊葡鉊Ｒ?鉊?鉊丞虜鉆腹
  ]);
}

/* 鉆鉊?葭鉊Ｒ? Mask / Accessory sheet: Active (In) 鉊?鉊冢? ??Out 鉆艇鉆艇鉊腹鉊耜?鉆葡鉊艇鉆葡鉊?*/
function sortMaskAccSheet(sheet) {
  const last = sheet.getLastRow();
  if (last < DATA_START + 1) return;
  const range = sheet.getRange(DATA_START, 1, last - DATA_START + 1, sheet.getLastColumn());
  range.sort([
    { column: COL_STATUS, ascending: true },  // "Active" < "Out" ??Active 鉊葆鉆?鉊?鉊冢?
    { column: 2,          ascending: true }   // 鉆鉊?葭鉊Ｒ?鉊葡鉊?Date In 鉊葡鉊Ｒ?鉊?鉊?鉊丞萼鉊艇鉊詮?鉊?
  ]);
}

function _needsSort(sheet, statusCol, dateCol) {
  const last = sheet.getLastRow();
  if (last < DATA_START + 1) return false;
  const statuses = sheet.getRange(DATA_START, statusCol, last - DATA_START + 1, 1).getValues();
  let seenOut = false;
  for (let i = 0; i < statuses.length; i++) {
    const status = normalizeText(statuses[i][0]);
    if (status === 'Out') seenOut = true;
    if (status === 'Active' && seenOut) return true;
  }
  return false;
}

function doStockOutByRow(ss, b, date, time) {
  const sheet = ss.getSheetByName(TABS[b.type]);
  if (!sheet) throw new Error('Sheet not found: ' + b.type);

  const r = parseInt(b.sheetRow, 10);
  if (!r || r < DATA_START || r > sheet.getLastRow()) {
    throw new Error('Invalid row');
  }
  const numCols = sheet.getLastColumn();
  const srcRow = sheet.getRange(r, 1, 1, numCols).getValues()[0];

  if (!rowMatchesPayload(srcRow, b.type, b)) {
    throw new Error('Row does not match scanned item');
  }

  const outQty = Math.max(1, parseInt(b.qty || 1, 10));

  if (b.type === 'device') {
    markOut(sheet, r, b.type, date);
    if (b.location || b.note) {
      const oldNote = normalizeText(srcRow[15]);
      const extra = (b.location ? 'OUT TO: ' + b.location : '') + (b.note ? ' | ' + b.note : '');
      sheet.getRange(r, 16).setValue(oldNote ? oldNote + ' | ' + extra : extra);
    }
    // sort 鉊徇艇鉊晤? appendHistory 鉆鉊虞鉆葉鉊?鉊冢?鉊萵鉊?row index 鉆鉊艇鉊菽?鉊Ｒ?
  } else {
    // Mask / Accessory: 鉊丞? Qty 鉊葡鉊﹤?鉊喪?鉊抉?鉊腦鉊毯? + 鉊芹腦鉆葡鉊?Out row
    const qtyCol = b.type === 'mask' ? MASK_COL_QTY : ACC_COL_QTY;
    const qty = parseInt(srcRow[qtyCol - 1] || 0, 10);
    if (qty <= 0) throw new Error('No stock left');
    const actualOutQty = Math.min(outQty, qty);

    if (qty > actualOutQty) {
      sheet.getRange(r, qtyCol).setValue(qty - actualOutQty);
    } else {
      markOut(sheet, r, b.type, date);
    }
    appendOutRow(sheet, b.type, srcRow, date, time, b.recipient || b.location || '', b.note || '', actualOutQty);
    b.qty = actualOutQty;
    b.beforeQty = qty;
    b.afterQty = Math.max(0, qty - actualOutQty);
    if (qty <= actualOutQty && _needsSort(sheet, COL_STATUS, 2)) sortMaskAccSheet(sheet);
  }

  appendHistory(ss, b, date, time, 'Stock Out');
}

function doBatchStocktake(ss, items, date, time, ctx) {
  if (!Array.isArray(items) || !items.length) throw new Error('No stocktake items');
  if (!String(ctx.operator || ctx.staffName || '').trim()) throw new Error('Missing operator');
  const products = readProductDB(ss);
  const byCode = new Map(products.map(p => [String(p.product_code || '').trim().toUpperCase(), p]));
  let updated = 0;

  items.forEach(item => {
    const code = String(item.productCode || item.product_code || '').trim().toUpperCase();
    const qty = parseInt(item.qty, 10);
    if (!code || !Number.isFinite(qty) || qty < 0) throw new Error('Invalid stocktake item: ' + code);
    const product = byCode.get(code);
    if (!product) throw new Error('Product code not found: ' + code);
    if (product.type === 'device') return;

    const sheet = ss.getSheetByName(TABS[product.type]);
    if (!sheet) throw new Error('Sheet not found: ' + product.type);
    const result = setAggregateQty_(sheet, product, qty, date, ctx);
    appendHistory(ss, {
      type: product.type, name: product.name, brand: product.brand, model: product.model, size: product.size,
      ref: product.ref, productCode: product.product_code, qty: qty, beforeQty: result.beforeQty, afterQty: result.afterQty,
      purpose: 'ปรับยอดจาก stocktake', recipient: '', user: ctx.user, operator: ctx.operator, note: 'batch stocktake'
    }, date, time, 'Stocktake');
    updated++;
  });
  return {updated:updated};
}

function setAggregateQty_(sheet, product, qty, date, ctx) {
  const type = product.type;
  const rowPayload = {
    type:type, name:product.name, brand:product.brand, model:product.model, size:product.size, ref:product.ref,
    lot:'', gtin:product.gtin || '', qty:qty, location:'Stocktake', note:'stocktake by ' + (ctx.operator || ctx.user || '')
  };
  const row = findExistingAggregateRow(sheet, type, rowPayload);
  const qtyCol = type === 'mask' ? MASK_COL_QTY : ACC_COL_QTY;
  if (row !== -1) {
    const beforeQty = parseInt(sheet.getRange(row, qtyCol).getValue() || 0, 10);
    sheet.getRange(row, qtyCol).setValue(qty);
    sheet.getRange(row, COL_STATUS).setValue(qty > 0 ? 'Active' : 'Out');
    styleDataRow(sheet, row, type);
    return {row:row, beforeQty:beforeQty, afterQty:qty};
  }
  if (qty <= 0) return {row:-1, beforeQty:0, afterQty:0};
  const seq = sheet.getLastRow() - HDR_ROW + 1;
  const newRow = type === 'mask'
    ? [seq,date,'',product.name,'Active',product.brand,product.model,product.size,product.ref,'',product.gtin || '','',qty,'stocktake create','']
    : [seq,date,'',product.name,'Active',product.brand,'',product.gtin || '','',qty,'Stocktake','stocktake create'];
  sheet.appendRow(newRow);
  const r = sheet.getLastRow();
  styleDataRow(sheet, r, type);
  return {row:r, beforeQty:0, afterQty:qty};
}

const ALERT_LOW_QTY  = 3;  // 鉆?鉆?鉆鉊虞鉊冢?鉆鉊﹤虞鉆葉 qty ??鉊?鉊耜?鉊菽?

function setupAlertTrigger() {
  // 鉊丞? trigger 鉆鉊?鉊耜?鉊菽?鉊虞鉆葉 checkAndAlert 鉊?鉊冢? (鉊?鉊冢?鉊萵鉊?鉆董)
  ScriptApp.getProjectTriggers()
    .filter(t => t.getHandlerFunction() === 'checkAndAlert')
    .forEach(t => ScriptApp.deleteTrigger(t));

  ScriptApp.newTrigger('checkAndAlert')
    .timeBased()
    .everyDays(1)
    .atHour(8)
    .inTimezone(TZ)
    .create();

  Logger.log('Trigger created: checkAndAlert daily at 08:00');
}

/** 鉊萵鉊?鉆?鉊晤?鉊徇艇鉊晤? ??鉊號鉊?鉊?葭鉊Ｒ?鉆?鉊?Trigger 鉊虜鉊葷鉊晤? */
function checkAndAlert() {
  const email = ALERT_EMAIL.trim();
  if (!email) { Logger.log('Missing ALERT_EMAIL'); return; }

  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const today = new Date(); today.setHours(0,0,0,0);
  const { lowItems, expItems } = _collectAlertItems(ss, today);

  if (!lowItems.length && !expItems.length) {
    Logger.log('No alert items found');
    return;
  }

  // 鉊?鉊冢?鉊萵鉊葵鉆?鉊?鉊???鉊芹?鉊?鉊徇腹鉆?鉊?鉊耜萼鉆鉊﹤虞鉆葉鉊芹?鉊耜?鉊葡鉊??鉆?鉊艇鉊菽?鉊Ｒ?
  const props      = PropertiesService.getScriptProperties();
  const currentKey = JSON.stringify({
    low: lowItems.map(x => x.name + '|' + x.size),
    exp: expItems.map(x => x.sn  + '|' + x.expDate),
  });
  if (currentKey === (props.getProperty('lastAlertKey') || '')) {
    Logger.log('Alerts already sent today; skipping duplicate send');
    return;
  }
  props.setProperty('lastAlertKey', currentKey);

  const subject = `Skyline Stock Alert - ${lowItems.length} low stock, ${expItems.length} near expiry`;
  MailApp.sendEmail({ to: email, subject, htmlBody: _buildAlertHtml(lowItems, expItems, today) });
  Logger.log('Stock alert sent to ' + email);
}

/** 鉊?葷鉊腦鉊抉腹鉊?葡鉊Ｒ?鉊耜腦鉊葭鉆?鉆葉鉊?鉊?鉊?鉊虞鉊冢? */
function _collectAlertItems(ss, today) {
  const lowItems = [];
  const expItems = [];
  const warnMs   = ALERT_EXP_DAYS * 86400000;

  // ?? Mask: 鉊腦鉊抉? low stock ??????????????????????????????
  const maskSheet = ss.getSheetByName(TABS.mask);
  if (maskSheet && maskSheet.getLastRow() >= DATA_START) {
    const data = maskSheet.getRange(
      DATA_START, 1,
      maskSheet.getLastRow() - DATA_START + 1,
      maskSheet.getLastColumn()
    ).getValues();
    data.forEach(row => {
      if (String(row[COL_STATUS - 1] || '').trim() !== 'Active') return;
      const qty = Number(row[MASK_COL_QTY - 1] || 0);
      if (qty <= ALERT_LOW_QTY) {
        lowItems.push({
          type: 'Mask',
          name: String(row[3] || ''),
          size: String(row[MASK_COL_SIZE - 1] || ''),
          qty,
        });
      }
    });
  }

  // ?? Mask: 鉊腦鉊抉? near expiry (col 15 = Expiry) ??????????
  // 鉊?葉鉊腦鉊晤?鉊萵鉆? sheet 鉆鉊?鉊?(鉆腹鉆腹鉊?col 15) 鉆艇鉊?sheet 鉆葦鉊﹤?
  if (maskSheet && maskSheet.getLastRow() >= DATA_START) {
    const numColsMask = maskSheet.getLastColumn();
    if (numColsMask >= MASK_COL_EXPIRY) {
      const maskData = maskSheet.getRange(
        DATA_START, 1,
        maskSheet.getLastRow() - DATA_START + 1,
        numColsMask
      ).getValues();
      maskData.forEach(row => {
        if (String(row[COL_STATUS - 1] || '').trim() !== 'Active') return;
        const expRaw = row[MASK_COL_EXPIRY - 1];
        if (!expRaw) return;
        const expDate = (expRaw instanceof Date) ? expRaw : new Date(expRaw);
        if (isNaN(expDate.getTime())) return;
        expDate.setHours(0,0,0,0);
        const daysLeft = Math.ceil((expDate - today) / 86400000);
        if (daysLeft <= ALERT_EXP_DAYS) {
          expItems.push({
            type:    'Mask',
            name:    String(row[3] || ''),
            sn:      String(row[MASK_COL_LOT - 1] || ''),  // 鉆?鉆?LOT 鉆?鉊?SN
            daysLeft,
            expDate: Utilities.formatDate(expDate, TZ, 'dd/MM/yyyy'),
          });
        }
      });
    }
  }

  // ?? Accessory: 鉊腦鉊抉? low stock ?????????????????????????
  const accSheet = ss.getSheetByName(TABS.accessory);
  if (accSheet && accSheet.getLastRow() >= DATA_START) {
    const data = accSheet.getRange(
      DATA_START, 1,
      accSheet.getLastRow() - DATA_START + 1,
      accSheet.getLastColumn()
    ).getValues();
    data.forEach(row => {
      if (String(row[COL_STATUS - 1] || '').trim() !== 'Active') return;
      const qty = Number(row[ACC_COL_QTY - 1] || 0);
      if (qty <= ALERT_LOW_QTY) {
        lowItems.push({
          type: 'Accessory',
          name: String(row[3] || ''),
          size: '',
          qty,
        });
      }
    });
  }

  // ?? Device: 鉊腦鉊抉? near expiry ??????????????????????????
  const devSheet = ss.getSheetByName(TABS.device);
  if (devSheet && devSheet.getLastRow() >= DATA_START) {
    const data = devSheet.getRange(
      DATA_START, 1,
      devSheet.getLastRow() - DATA_START + 1,
      devSheet.getLastColumn()
    ).getValues();
    data.forEach(row => {
      if (String(row[COL_STATUS - 1] || '').trim() !== 'Active') return;
      const expRaw = row[11]; // col 12 = Expiry
      if (!expRaw) return;
      const expDate = new Date(expRaw);
      if (isNaN(expDate.getTime())) return;
      expDate.setHours(0,0,0,0);
      const daysLeft = Math.ceil((expDate - today) / 86400000);
      if (daysLeft <= ALERT_EXP_DAYS) {
        expItems.push({
          type:    'Device',
          name:    String(row[3] || ''),
          sn:      String(row[DEV_COL_SN - 1] || ''),
          daysLeft,
          expDate: Utilities.formatDate(expDate, TZ, 'dd/MM/yyyy'),
        });
      }
    });
  }

  return { lowItems, expItems };
}

/** 鉊芹腦鉆葡鉊?HTML body 鉊葉鉊葉鉊菽?鉊﹤艇鉆?鉆?鉆鉊虞鉊冢? */
function _buildAlertHtml(lowItems, expItems, today) {
  const dateStr = Utilities.formatDate(today, TZ, 'dd/MM/yyyy');

  const lowRows = lowItems.map(it => {
    const color = it.qty === 0 ? '#b92e2e' : '#a06a00';
    const label = it.qty === 0 ? '0 (หมดสต็อก)' : String(it.qty);
    return `<tr>
      <td style="padding:8px 12px;border-bottom:1px solid #f0f0f0">${it.type}</td>
      <td style="padding:8px 12px;border-bottom:1px solid #f0f0f0">${it.name}${it.size ? ' (' + it.size + ')' : ''}</td>
      <td style="padding:8px 12px;border-bottom:1px solid #f0f0f0;color:${color};font-weight:700">${label}</td>
    </tr>`;
  }).join('');

  const expRows = expItems.map(it => {
    const color = it.daysLeft <= 0 ? '#b92e2e' : it.daysLeft <= 30 ? '#b92e2e' : '#a06a00';
    const label = it.daysLeft <= 0 ? 'Expired' : ('Days left: ' + it.daysLeft);
    return `<tr>
      <td style="padding:8px 12px;border-bottom:1px solid #f0f0f0">${it.name}</td>
      <td style="padding:8px 12px;border-bottom:1px solid #f0f0f0;font-family:monospace">${it.sn || '-'}</td>
      <td style="padding:8px 12px;border-bottom:1px solid #f0f0f0">${it.expDate}</td>
      <td style="padding:8px 12px;border-bottom:1px solid #f0f0f0;color:${color};font-weight:700">${label}</td>
    </tr>`;
  }).join('');

  const lowSection = lowItems.length ? `
    <h3 style="margin:24px 0 8px;color:#a06a00;font-size:14px">Low stock: ${lowItems.length} items (threshold: ${ALERT_LOW_QTY})</h3>
    <table style="width:100%;border-collapse:collapse;font-size:13px;background:#fff">
      <thead><tr style="background:#fdf3d0;text-align:left">
        <th style="padding:8px 12px">Type</th>
        <th style="padding:8px 12px">Product</th>
        <th style="padding:8px 12px">Qty</th>
      </tr></thead>
      <tbody>${lowRows}</tbody>
    </table>` : '';

  const expSection = expItems.length ? `
    <h3 style="margin:24px 0 8px;color:#b92e2e;font-size:14px">Near expiry: ${expItems.length} items (within ${ALERT_EXP_DAYS} days)</h3>
    <table style="width:100%;border-collapse:collapse;font-size:13px;background:#fff">
      <thead><tr style="background:#fde8e8;text-align:left">
        <th style="padding:8px 12px">Product</th>
        <th style="padding:8px 12px">Serial Number</th>
        <th style="padding:8px 12px">Expiry</th>
        <th style="padding:8px 12px">Status</th>
      </tr></thead>
      <tbody>${expRows}</tbody>
    </table>` : '';

  return `
<!DOCTYPE html>
<html><head><meta charset="UTF-8"></head>
<body style="margin:0;padding:20px;background:#f0f2f5;font-family:Arial,sans-serif">
  <div style="max-width:600px;margin:0 auto">
    <div style="background:#1a1a2e;color:#fff;padding:18px 22px;border-radius:10px 10px 0 0">
      <div style="font-size:18px;font-weight:700">Skyline Medical Stock Alert</div>
      <div style="font-size:12px;opacity:.65;margin-top:4px">Stock alert summary - ${dateStr}</div>
    </div>
    <div style="background:#fff;padding:20px 22px;border:1px solid #ddd;border-top:none;border-radius:0 0 10px 10px">
      ${lowSection}
      ${expSection}
      <div style="margin-top:24px;padding-top:16px;border-top:1px solid #eee;font-size:11px;color:#999;line-height:1.7">
        This alert was generated automatically by Skyline Stock.<br>
        Low-stock threshold: ${ALERT_LOW_QTY} items &nbsp;|&nbsp; Expiry warning: ${ALERT_EXP_DAYS} days<br>
        To change recipients, update ALERT_EMAIL in Apps Script.
      </div>
    </div>
  </div>
</body></html>`;
}

function checkDuplicate(ss, type, sn, lot, name, size) {
  return !!findDuplicate(ss, type, sn, lot, name, size);
}

function findDuplicate(ss, type, sn, lot, name, size) {
  const sheet = ss.getSheetByName(TABS[type]);
  if (!sheet) return null;
  const last = sheet.getLastRow();
  if (last < DATA_START) return null;

  const numCols = sheet.getLastColumn();
  const data = sheet.getRange(DATA_START, 1, last - DATA_START + 1, numCols).getValues();

  for (let i = 0; i < data.length; i++) {
    const row = data[i];
    const status = normalizeText(row[COL_STATUS - 1]);
    if (status !== 'Active') continue;

    if (type === 'device') {
      if (sn) {
        // 鉊腦鉊抉?鉊?鉊喪?鉆葷鉊?SN (鉊腦鉊葭鉊?鉊葩)
        const rowSn = normalizeText(row[DEV_COL_SN - 1]);
        if (rowSn.toUpperCase() === normalizeText(sn).toUpperCase()) {
          return {
            row: i + DATA_START,
            status: status,
            sn: rowSn,
            lot: normalizeText(row[DEV_COL_LOT - 1]),
            name: normalizeText(row[3])
          };
        }
      }
      // 鉊?鉊?barcode 鉆腹鉆腹鉊?SN 鉊冢腺鉆葡鉊萵鉊?鉆董鉊?鉊抉腺 LOT/GTIN 鉆鉊腦鉊耜萼鉆腹鉆?鉊?鉊?葦鉊晤葵鉊?葡鉊Ｒ?鉊腦鉊獅?鉊冢?
      // 鉆艇鉊啤?鉊啤?鉊喪?鉊徇?鉆鉊腦鉊獅?鉊冢?鉊葭鉆?鉊腺 Out 鉆艇鉆葷鉊?萵鉊?鉊丞萵鉊?鉊?鉊耜?鉊﹤?鉆?鉆?鉊﹤虞鉆葉鉊Ｒ萵鉊腹鉊菽?鉊腦鉊獅?鉊冢? LOT 鉆鉊葭鉊Ｒ葷鉊萵鉊?Active 鉊冢腺鉊嫩?
    }

    // mask/accessory: 鉆腹鉆?鉊?葷鉊?鉆董鉊葭鉆?鉊菽? ??same LOT 鉊葷鉊?aggregate qty 鉆? doStockIn()
    // 鉊萵鉊?鉊晤?鉊?duplicate check 鉆?鉆?鉊?鉊耜萼 device 鉆鉊?鉊耜?鉊晤?鉊?
  }
  return null;
}
